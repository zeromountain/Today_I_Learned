# 일급함수
- 함수를 값으로 다룰 수 있다
- 조합성과 추상화의 도구 : 

```js
  const add5 = a => a + 5;
  console.log(add5); // 함수를 값으로 다룸, 함수의 인자로 사용될 수 있음 
  console.log(add5(5));

  const foo = () => () => 1;
  console.log(foo()); // () => 1;
  
  const foo2 = foo();
  console.log(f2); // () => 1;
  console.log(f2()); // 1
  ```
