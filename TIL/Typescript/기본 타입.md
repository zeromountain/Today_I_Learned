# 타입스크립트에서의 기본 타입 정의

- String
```js
// JS에서 변수에 문자열 할당
let str = 'hello';
```
```ts
// TS에서 변수에 문자열 할당
let str: string = 'hello';
```
타입스크립트에서는 변수 뒤에 `: type`을 통해서 변수의 타입을 정해 줄 수 있다.

- Number
```ts
let num: number = 10;
```

- Array
```ts
// 배열 요소의 타입을 먼저 써주고 배열 리터럴을 이용해 기본적인 배열을 선언 할 수 있다.
// 요소에 number 이외의 타입이 들어가면 오류가 발생한다.
let arr: number[]= [1, 2, 3];

// 제네릭을 활용한 변수에 배열 할당 방법
let arr: Array<number> = [1, 2, 3];

// Tuple을 활용하기
// Tuple: 배열의 특정 인덱스 요소의 타입을 정의하는 방법
// 첫번째는 1번 인덱스에 문자 타입이 들어갔기 때문에 오류가 발생한다.
let address: [string, number] = ['gangnam-ro', 'pangyo-ro'];
address = ['gangnam-ro', 10];
```

- Object
```ts
// 기본적인 객체 타입 정의
let obj: object = {};
// 객체의 key value에 타입을 정의하지 않은 경우
let capt: object = {
  name: 'capt',
  age: 100
}
// 객체의 key value에 타입을 정의한 경우
let thor: {name: string, age: number} = {
  name: 'thor',
  age: 1000
}
```

- Boolean
```ts
let show: booean = true;
```
